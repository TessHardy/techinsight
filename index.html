function selectScene(scene){
  currentScene=scene;
  document.querySelectorAll('.scene-btn').forEach(function(btn){
    btn.classList.toggle('active',btn.getAttribute('data-scene')===scene);
  });
  
  // 计算各框架得分
  var weights=sceneData[scene].weights;
  var scores=[];
  var frameworks=['React','Vue','Angular','Svelte','Next.js','Nuxt'];
  var colors=['#3b82f6','#10b981','#f59e0b','#8b5cf6','#06b6d4','#ec4899'];
  
  frameworks.forEach(function(fw,i){
    var total=0;
    for(var dim in weights){
      var fwScore=frameworkScores[fw][dim]||0;
      total+=fwScore*weights[dim];
    }
    scores.push({
      name:fw,
      score:Math.round(total),
      color:colors[i]
    });
  });
  
  // 排序
  scores.sort(function(a,b){return b.score-a.score});
  
  // 更新图表
  scoreChartInstance.setOption({
    backgroundColor:currentTheme==='dark'?'transparent':'#ffffff',
    tooltip:{trigger:'axis',formatter:function(p){return p[0].name+'
综合评分：'+p[0].value+'分'}},
    grid:{left:'3%',right:'4%',bottom:'3%',containLabel:true},
    xAxis:{type:'category',data:scores.map(function(s){return s.name}),axisLine:{lineStyle:{color:currentTheme==='dark'?'rgba(255,255,255,0.1)':'rgba(0,0,0,0.1)'}}},
    yAxis:{type:'value',max:100,name:'综合评分',axisLine:{lineStyle:{color:currentTheme==='dark'?'rgba(255,255,255,0.1)':'rgba(0,0,0,0.1)'}}},
    series:[{
      type:'bar',
      data:scores.map(function(s){return {value:s.score,itemStyle:{color:s.color,borderRadius:[6,6,0,0]}}}),
      barWidth:40,
      label:{show:true,position:'top',formatter:'{c}分',color:currentTheme==='dark'?'#e2e8f0':'#1e293b'}
    }],
    animationDuration:800
  });
  
  // 更新推荐
  var top=scores[0];
  var second=scores[1];
  var recommendText='针对'+sceneData[scene].name+'场景（'+sceneData[scene].desc+'），推荐：

';
  recommendText+=''+top.name+' 综合得分 '+top.score+' 分，是最佳选择。
';
  recommendText+='备选：'+second.name+'（'+second.score+' 分）。

';
  
  // 添加场景特定建议
  if(scene==='admin'){
    recommendText+='💡 中后台系统优先考虑开发体验和生态成熟度，React 的生态最丰富，适合复杂表单和权限管理。';
  }else if(scene==='frontend'){
    recommendText+='💡 大前端项目优先考虑性能和用户体验，Svelte 性能最优，React 生态最成熟。';
  }else if(scene==='mobile'){
    recommendText+='💡 移动端开发关注包体积和性能，Svelte 包体积最小（仅8KB），是最优选择。';
  }else if(scene==='fullstack'){
    recommendText+='💡 全栈应用需要 SEO 和服务端渲染，Next.js 是 React 生态的最佳方案。';
  }else if(scene==='seo'){
    recommendText+='💡 SEO 优化场景，Next.js 和 Nuxt 提供最佳的服务端渲染支持。';
  }else if(scene==='startup'){
    recommendText+='💡 初创项目需要快速上线，Vue 学习曲线最友好，开发效率最高。';
  }
  
  document.getElementById('scoreRecommendText').innerHTML=recommendText;
}

function toggleFilter(el){
  var fw=el.getAttribute('data-framework');
  activeFilters[fw]=!activeFilters[fw];
  if(activeFilters[fw]){
    el.classList.add('active');
    el.classList.remove('inactive');
  }else{
    el.classList.remove('active');
    el.classList.add('inactive');
  }
  updateAllCharts();
}

function switchTimeRange(years,btn){
  currentTimeRange=years;
  document.querySelectorAll('.time-btn').forEach(function(b){b.classList.remove('active')});
  btn.classList.add('active');
  updateTrendChart();
}

function getFilteredFrameworks(){
  var fws=[];
  for(var k in activeFilters){if(activeFilters[k])fws.push(k)}
  return fws;
}

var frameworkColors={React:'#3b82f6',Vue:'#10b981',Angular:'#f59e0b',Svelte:'#8b5cf6','Next.js':'#06b6d4',Nuxt:'#ec4899'};

function updateAllCharts(){
  updateTrendChart();
  updateCompareChart();
  updatePieChart();
  updateRadarChart();
  updateScatterChart();
}

function updateTrendChart(){
  var chart=chartInstances.trend;
  if(!chart)return;
  var startIdx=trendData.years.length-currentTimeRange;
  var years=trendData.years.slice(startIdx);
  var series=[];
  ['React','Vue','Angular','Svelte'].forEach(function(fw){
    if(!activeFilters[fw])return;
    var color=frameworkColors[fw];
    var data=trendData[fw].slice(startIdx);
    series.push({name:fw,type:'line',smooth:true,data:data,lineStyle:{width:3},itemStyle:{color:color},areaStyle:{color:new echarts.graphic.LinearGradient(0,0,0,1,[{offset:0,color:color},{offset:1,color:color}])},symbol:'circle',symbolSize:6});
  });
  chart.setOption({xAxis:{data:years},legend:{data:series.map(function(s){return s.name})},series:series});
}

function updateCompareChart(){
  var chart=chartInstances.compare;
  if(!chart)return;
  var fws=getFilteredFrameworks();
  var cats=[],stars=[],downloads=[],issueTime=[];
  compareData.categories.forEach(function(c,i){
    if(fws.indexOf(c)!==-1){
      cats.push(c);stars.push(compareData.stars[i]);
      downloads.push(compareData.downloads[i]);issueTime.push(compareData.issueTime[i]);
    }
  });
  chart.setOption({
    xAxis:{data:cats},
    legend:{data:cats.length?['GitHub Stars(K)','NPM周下载(万)','Issue响应时间(h)']:[]},
    series:[
      {name:'GitHub Stars(K)',type:'bar',data:stars,itemStyle:{color:'#3b82f6',borderRadius:[4,4,0,0]},barWidth:20},
      {name:'NPM周下载(万)',type:'bar',data:downloads,itemStyle:{color:'#8b5cf6',borderRadius:[4,4,0,0]},barWidth:20},
      {name:'Issue响应时间(h)',type:'line',yAxisIndex:1,data:issueTime,lineStyle:{width:3,color:'#f59e0b'},itemStyle:{color:'#f59e0b'},symbol:'circle',symbolSize:8}
    ]
  });
}

function updatePieChart(){
  var chart=chartInstances.pie;
  if(!chart)return;
  var fws=getFilteredFrameworks();
  var data=pieData.filter(function(d){return fws.indexOf(d.name)!==-1||d.name==='其他'});
  chart.setOption({series:[{data:data}]});
}

function updateRadarChart(){
  var chart=chartInstances.radar;
  if(!chart)return;
  var series=[];
  ['React','Vue','Svelte'].forEach(function(fw){
    if(!activeFilters[fw])return;
    var color=frameworkColors[fw];
    series.push({value:radarData[fw],name:fw,lineStyle:{color:color,width:2},areaStyle:{color:color},itemStyle:{color:color}});
  });
  chart.setOption({legend:{data:series.map(function(s){return s.name})},series:[{type:'radar',data:series}]});
}

function updateScatterChart(){
  var chart=chartInstances.scatter;
  if(!chart)return;
  var fws=getFilteredFrameworks();
  var data=scatterData.filter(function(d){return fws.indexOf(d[3])!==-1});
  chart.setOption({series:[{data:data}]});
}

function askAi(question){

  typing.id='aiTyping';
  typing.innerHTML='';
  msgs.appendChild(typing);
  msgs.scrollTop=msgs.scrollHeight;

  var answer=aiKnowledge[question]||aiKnowledge['default'];
  setTimeout(function(){
    var t=document.getElementById('aiTyping');
    if(t)t.remove();

    botMsg.className='ai-msg bot';
    msgs.appendChild(botMsg);
    typeWriterEffect(botMsg,answer,0);
  },800+Math.random()*600);
}

  if(idx>=text.length){
    var msgs=document.getElementById('aiMessages');

    return;
  }
  var ch=text[idx];
  if(ch==='\n'){el.innerHTML+='
'}
  else{el.innerHTML+=ch}
  var msgs=document.getElementById('aiMessages');
  msgs.scrollTop=msgs.scrollHeight;
  setTimeout(function(){typeWriterEffect(el,text,idx+1)},ch==='\n'?80:18);
}

/* ========== 数字动画 ========== */
(function(){
  var animated=false;
  function animateNumbers(){
    document.querySelectorAll('.stat-number').forEach(function(el){
      var target=parseInt(el.getAttribute('data-target'));
      var suffix=el.getAttribute('data-suffix')||'';
      var duration=2000,startTime=null;
      function step(ts){
        if(!startTime)startTime=ts;
        var progress=Math.min((ts-startTime)/duration,1);
        var eased=1-Math.pow(1-progress,3);
        el.textContent=Math.floor(eased*target).toLocaleString()+suffix;
        if(progress<1)requestAnimationFrame(step);
      }
      requestAnimationFrame(step);
    });
  }
  var observer=new IntersectionObserver(function(entries){
    entries.forEach(function(entry){
      if(entry.isIntersecting&&!animated){animated=true;animateNumbers()}
    });
  },{threshold:0.3});
  observer.observe(document.getElementById('stats'));
})();

      }
    });
  },{threshold:0.1});
  document.querySelectorAll('.chart-card,.feature-card,.case-card').forEach(function(card){observer.observe(card)});
})();
